def notify_private_reply(m, message_type = 'your interaction')
  if m.channel?
    m.reply(message_type + ' continues via PM', true)
  end
end

def reply_in_pm(m, reply_lines = [], &block)
  if block_given?
    reply_lines = yield(reply_lines)
  end
  reply_lines.each { |l| m.user.msg(l) }
end
