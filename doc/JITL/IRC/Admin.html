def self.authenticate(username, password)
  if @@Admin_Users.include? username
    if @@Admin_Users[username][:password] == password
      @@Admin_Users[username][:auth] = Auth::AUTHED
      return Auth::AUTHED
    else
      return Auth::INVALID_PASSOWRD_ERROR
    end
  else
    return Auth::INVALID_USER_ERROR
  end
end

def admin_add(m, username, password)
  if Admin.authed? m.user.nick
    if username && password
      Admin.add_admin(username, password) if username && password
      m.reply("added #{username} as an admin for this bot", true)
    else
      m.reply("needs username and password")
    end
  else
    no_permission(m)
  end
end

def admin_remove(m, username)

def auth_password(m, password)
  auth_status = Admin.authenticate(m.user.nick, password)
  if auth_status == Auth::AUTHED
    m.reply('you have been authenticated.')
  else
    m.reply('Invalid user or password')
  end
end

def listen(m)
  if Admin.is_admin? m.user.last_nick
    m.reply("Your name changed from #{m.user.last_nick} to #{m.user.nick}")
    m.reply("I am persisting your admin status onto new nickname #{m.user.nick}")
    Admin.change_username(m.user.last_nick, m.user.nick)
  end
end

def tell_all(m, message)
  if Admin.authed? m.user.nick
    @bot.channels.each {|c|
      c.send(message)
    }
  else
    no_permission m
  end
end
